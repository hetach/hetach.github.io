84 void WebServerPrivate::handler(evhttp_request *req, void *)
85 {
86  evbuffer *outBuffer = evhttp_request_get_output_buffer(req);
87 
88  if(!outBuffer) {
89  return;
90  }
91 
92  Request *request = Request::create(req);
93 
94  Response *response = WebServerPrivate::onRequest(request);
95 
96  evkeyvalq *outputHeaders = evhttp_request_get_output_headers(req);
97 
98  string version = HETACH_VERSION;
99  response->addHeader(Header("Server", "hetach/" + version));
100 
101  response->addHeader(Header("Connection", "close"));
102 
103  vector<Header> headers = response->headers();
104 
105  for(vector<Header>::iterator it = headers.begin(); it != headers.end(); ++it) {
106  Header header = static_cast<Header>(*it);
107 
108  evhttp_add_header(outputHeaders, header.name().data(), header.value().data());
109  }
110 
111  evbuffer_add(outBuffer, response->content().data(), response->content().size());
112 
113  evhttp_send_reply(req, response->statusCode(), "", outBuffer);
114 
115  delete request;
116  delete response;
117 }
