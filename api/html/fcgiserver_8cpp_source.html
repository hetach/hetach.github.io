41 bool FCGIServer::listen()
42 {
43  this->m_fcgiRequest = new FCGX_Request();
44 
45  if(FCGX_Init() != 0) {
46  cerr << "FCGX_Init failed" << endl;
47  return false;
48  }
49 
50  if(FCGX_InitRequest(this->m_fcgiRequest, 0, 0) != 0) {
51  cerr << "FCGX_InitRequest failed" << endl;
52  return false;
53  }
54 
55  while(FCGX_Accept_r(this->m_fcgiRequest) == 0) {
56  Request *request = Request::create(this->m_fcgiRequest);
57 
58  Response *response = this->m_onRequest(request);
59 
60  string status = Header("Status", response->statusCode()).toString() + "\r\n";
61 
62  FCGX_PutStr(status.data(), status.size(), this->m_fcgiRequest->out);
63 
64  vector<Header> headers = response->headers();
65 
66  for(vector<Header>::iterator it = headers.begin(); it != headers.end(); ++it) {
67  Header header = static_cast<Header>(*it);
68 
69  string data = header.toString() + "\r\n";
70 
71  FCGX_PutStr(data.data(), data.size(), this->m_fcgiRequest->out);
72  }
73 
74  string content = "\r\n" + response->content();
75 
76  FCGX_PutStr(content.data(), content.size(), this->m_fcgiRequest->out);
77 
78  delete request;
79  delete response;
80  }
81 
82  return true;
83 }
